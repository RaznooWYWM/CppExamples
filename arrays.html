// Arrays
        int[] my_array = {1, 3, 4,  6, 7, 8};
        my_array[0];      // equivalent to 1
        my_array[2];       // equivalent to 4
        
        my_array[2] = 5;
        my_array;          // equivalent to  [1, 3, 5,  6, 7, 8]

        list<int> my_list;    //starts as an empty list
        my_list.push_back(3);    // add the integer 3 to my_list
        my_list.push_back(5);    // add the integer 5 to my_list
        my_list;           // equivalent to (3, 5)
